/* ========== JEMNÉ OBJEVENÍ SEKCÍ PŘI SCROLLU ========== */

(function(){

  var prvky = document.querySelectorAll('.animace');

  if (!('IntersectionObserver' in window)){

    prvky.forEach(function(p){ p.classList.add('viditelne'); });

    return;

  }

  var pozorovatel = new IntersectionObserver(function(zaznamy){

    zaznamy.forEach(function(z){

      if (z.isIntersecting){

        z.target.classList.add('viditelne');

        pozorovatel.unobserve(z.target);

      }

    });

  }, { threshold: 0.12 });

  prvky.forEach(function(p){ pozorovatel.observe(p); });

})();
